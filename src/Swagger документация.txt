openapi: 3.0.0
info:
  title: Сообщения API
  version: 1.0.0
  description: API для работы с сообщениями, включая добавление и получение сообщений.
servers:
  - url: http://localhost:8082/Api/V1
paths:
  /AddMsg:
    post:
      summary: Добавление сообщения
      description: Метод для добавления нового сообщения в базу данных.
      parameters:
        - name: msg
          in: query
          required: true
          description: Текст сообщения, которое необходимо добавить.
          schema:
            type: string
      responses:
        '200':
          description: Сообщение успешно добавлено.
          content:
            application/json:
              schema:
                type: object
                properties:
                  success:
                    type: boolean
                    example: true
                  message:
                    type: string
                    example: "Сообщение успешно добавлено."
        '400':
          description: Неправильный запрос.
          content:
            application/json:
              schema:
                type: object
                properties:
                  success:
                    type: boolean
                    example: false
                  error:
                    type: string
                    example: "Некорректный ввод."

  /GetMsg:
    get:
      summary: Получение всех сообщений
      description: Метод для получения списка всех сообщений в базе данных.
      responses:
        '200':
          description: Успешное получение списка сообщений.
          content:
            application/json:
              schema:
                type: object
                properties:
                  messages:
                    type: array
                    items:
                      type: object
                      properties:
                        id:
                          type: string
                          example: "684c188c9446b"
                        msg:
                          type: string
                          example: "Привет"
        '500':
          description: Ошибка сервера при получении сообщений.

  /GetMsg/{id}:
    get:
      summary: Получение сообщения по ID
      description: Метод для получения сообщения по его уникальному идентификатору.
      parameters:
        - name: id
          in: path
          required: true
          description: Идентификатор сообщения, которое необходимо получить.
          schema:
            type: string
      responses:
        '200':
          description: Успешное получение сообщения.
          content:
            application/json:
              schema:
                type: object
                properties:
                  id:
                    type: string
                    example: "684c188c9446b"
                  msg:
                    type: string
                    example: "Привет"
        '404':
          description: Сообщение с данным ID не найдено.
          content:
            application/json:
              schema:
                type: object
                properties:
                  success:
                    type: boolean
                    example: false
                  error:
                    type: string
                    example: "Сообщение не найдено."
        '500':
          description: Ошибка сервера при получении сообщения.
